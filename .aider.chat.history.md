import sqlite3
import os

DB_PATH = "news_scraper.db"

def init_database():
    """Initialize the SQLite database with required tables."""
    conn = sqlite3.connect(DB_PATH)
    cursor = conn.cursor()

    # Create sites table
    cursor.execute('''
        CREATE TABLE IF NOT EXISTS sites (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            name TEXT UNIQUE NOT NULL,
            base_url TEXT NOT NULL,
            description TEXT,
            active INTEGER DEFAULT 1,
            created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
        )
    ''')

    # Create categories table
    cursor.execute('''
        CREATE TABLE IF NOT EXISTS categories (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            site_id INTEGER NOT NULL,
            name TEXT NOT NULL,
            url_path TEXT NOT NULL,
            description TEXT,
            FOREIGN KEY (site_id) REFERENCES sites(id)
        )
    ''')

    # Create pages table
    cursor.execute('''
        CREATE TABLE IF NOT EXISTS pages (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            category_id INTEGER NOT NULL,
            url TEXT NOT NULL,
            last_scraped TIMESTAMP,
            is_active INTEGER DEFAULT 1,
            FOREIGN KEY (category_id) REFERENCES categories(id)
        )
    ''')

    # Create articles table
    cursor.execute('''
        CREATE TABLE IF NOT EXISTS articles (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            site_id INTEGER NOT NULL,
            category_id INTEGER NOT NULL,
            title TEXT NOT NULL,
            url TEXT UNIQUE NOT NULL,
            summary TEXT,
            published_date TIMESTAMP,
            scraped_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
            FOREIGN KEY (site_id) REFERENCES sites(id),
            FOREIGN KEY (category_id) REFERENCES categories(id)
        )
    ''')

    conn.commit()
    conn.close()

def add_site(name, base_url, description=""):
    """Add a new news site to the database."""
    conn = sqlite3.connect(DB_PATH)
    cursor = conn.cursor()
    
    try:
        cursor.execute('''
            INSERT INTO sites (name, base_url, description)
            VALUES (?, ?, ?)
        ''', (name, base_url, description))
        conn.commit()
    except sqlite3.IntegrityError:
        print(f"Site '{name}' already exists.")
    finally:
        conn.close()

def add_category(site_id, name, url_path, description=""):
    """Add a category to a site."""
    conn = sqlite3.connect(DB_PATH)
    cursor = conn.cursor()
    
    cursor.execute('''
        INSERT INTO categories (site_id, name, url_path, description)
        VALUES (?, ?, ?, ?)
    ''', (site_id, name, url_path, description))
    conn.commit()
    conn.close()

def add_page(category_id, url):
    """Add a page URL to scrape."""
    conn = sqlite3.connect(DB_PATH)
    cursor = conn.cursor()
    
    cursor.execute('''
        INSERT INTO pages (category_id, url)
        VALUES (?, ?)
    ''', (category_id, url))
    conn.commit()
    conn.close()

if __name__ == "__main__":
    # Initialize the database
    init_database()
    print("Database initialized successfully.")

import sqlite3

DB_PATH = "news_scraper.db"

class ConfigManager:
    def __init__(self):
        self.conn = sqlite3.connect(DB_PATH)
        self.cursor = self.conn.cursor()

    def get_active_sites(self):
        """Get all active sites."""
        self.cursor.execute("SELECT id, name, base_url FROM sites WHERE active = 1")
        return self.cursor.fetchall()

    def get_categories_for_site(self, site_id):
        """Get categories for a specific site."""
        self.cursor.execute(
            "SELECT id, name, url_path FROM categories WHERE site_id = ?", 
            (site_id,)
        )
        return self.cursor.fetchall()

    def get_pages_for_category(self, category_id):
        """Get pages for a specific category."""
        self.cursor.execute(
            "SELECT id, url FROM pages WHERE category_id = ? AND is_active = 1", 
            (category_id,)
        )
        return self.cursor.fetchall()

    def update_scrape_timestamp(self, page_id):
        """Update the last scraped timestamp for a page."""
        self.cursor.execute(
            "UPDATE pages SET last_scraped = CURRENT_TIMESTAMP WHERE id = ?", 
            (page_id,)
        )
        self.conn.commit()

    def add_article(self, site_id, category_id, title, url, summary=None, published_date=None):

import requests
from bs4 import BeautifulSoup
from abc import ABC, abstractmethod

class BaseScraper(ABC):
    def __init__(self, base_url):
        self.base_url = base_url
        self.session = requests.Session()
        self.session.headers.update({
            'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/91.0.4472.124 Safari/537.36'
        })

    @abstractmethod
    def parse_latest_articles(self, page_content):
        """Parse latest articles from a page."""
        pass

    @abstractmethod
    def parse_historical_articles(self, page_content):
        """Parse historical articles from a page."""
        pass

    def fetch_page(self, url):
        """Fetch a web page and return its content."""
        try:
            response = self.session.get(url)
            response.raise_for_status()
            return response.text
        except requests.RequestException as e:
            print(f"Error fetching {url}: {e}")
            return None

    def scrape_latest(self, url):
        """Scrape latest articles from a URL."""
        content = self.fetch_page(url)
        if not content:
            return []
        
        return self.parse_latest_articles(content)

    def scrape_historical(self, url):
        """Scrape historical articles from a URL."""
        content = self.fetch_page(url)
        if not content:
            return []
        
        return self.parse_historical_articles(content)

        self.db_path = db_path
        self.conn = sqlite3.connect(db_path)
        self.cursor = self.conn.cursor()
        self._create_tables()

    def _create_tables(self):
        # Create websites table
        self.cursor.execute('''
            CREATE TABLE IF NOT EXISTS websites (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                name TEXT UNIQUE NOT NULL,
                url TEXT NOT NULL,
                description TEXT,
                active BOOLEAN DEFAULT 1,
                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
                updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
            )
        ''')

        # Create categories table
        self.cursor.execute('''
            CREATE TABLE IF NOT EXISTS categories (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                website_id INTEGER NOT NULL,
                name TEXT NOT NULL,
                url TEXT NOT NULL,
                category_type TEXT CHECK(category_type IN ('latest', 'historical')) DEFAULT 'latest',
                active BOOLEAN DEFAULT 1,
                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
                FOREIGN KEY (website_id) REFERENCES websites(id)
            )
        ''')

        # Create articles table
        self.cursor.execute('''
            CREATE TABLE IF NOT EXISTS articles (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                website_id INTEGER NOT NULL,
                category_id INTEGER,
                title TEXT NOT NULL,
                url TEXT UNIQUE NOT NULL,
                summary TEXT,
                published_date TIMESTAMP,
                scraped_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
                FOREIGN KEY (website_id) REFERENCES websites(id),
                FOREIGN KEY (category_id) REFERENCES categories(id)
            )
        ''')

        # Create scraping log table
        self.cursor.execute('''
            CREATE TABLE IF NOT EXISTS scraping_logs (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                website_id INTEGER NOT NULL,
                category_id INTEGER,
                mode TEXT CHECK(mode IN ('latest', 'historical')) NOT NULL,
                articles_found INTEGER DEFAULT 0,
                articles_saved INTEGER DEFAULT 0,
                status TEXT CHECK(status IN ('success', 'partial_success', 'failed')) DEFAULT 'success',
                error_message TEXT,
                started_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
                completed_at TIMESTAMP,
                FOREIGN KEY (website_id) REFERENCES websites(id),
                FOREIGN KEY (category_id) REFERENCES categories(id)
            )
        ''')

        self.conn.commit()

    def add_website(self, name: str, url: str, description: str = "") -> int:
        try:
            self.cursor.execute('''
                INSERT INTO websites (name, url, description)
                VALUES (?, ?, ?)
            ''', (name, url, description))
            self.conn.commit()
            return self.cursor.lastrowid
        except sqlite3.IntegrityError:
            # Website already exists, get its ID
            self.cursor.execute('SELECT id FROM websites WHERE name = ?', (name,))
            return self.cursor.fetchone()[0]

    def add_category(self, website_id: int, name: str, url: str, category_type: str = 'latest') -> int:
        try:
            self.cursor.execute('''
                INSERT INTO categories (website_id, name, url, category_type)
                VALUES (?, ?, ?, ?)
            ''', (website_id, name, url, category_type))
            self.conn.commit()
            return self.cursor.lastrowid
        except sqlite3.IntegrityError:
            # Category already exists, get its ID
            self.cursor.execute('''
                SELECT id FROM categories WHERE website_id = ? AND url = ?
            ''', (website_id, url))
            result = self.cursor.fetchone()
            if result:
                return result[0]
            raise

    def add_article(self, website_id: int, category_id: Optional[int], title: str, 
                   url: str, summary: str = "", published_date: datetime = None) -> bool:
        try:
            self.cursor.execute('''
                INSERT INTO articles (website_id, category_id, title, url, summary, published_date)
                VALUES (?, ?, ?, ?, ?, ?)
            ''', (website_id, category_id, title, url, summary, 
                  published_date.isoformat() if published_date else None))
            self.conn.commit()
            return True
        except sqlite3.IntegrityError:
            # Article already exists
            return False

    def log_scraping(self, website_id: int, category_id: Optional[int], mode: str,
                    articles_found: int, articles_saved: int, status: str = 'success',
                    error_message: str = "") -> int:
        self.cursor.execute('''

            VALUES (?, ?, ?, ?, ?, ?, ?)
        ''', (website_id, category_id, mode, articles_found, articles_saved, 
              status, error_message))
        self.conn.commit()
        return self.cursor.lastrowid

        self.cursor.execute('SELECT * FROM websites WHERE active = 1')

        """Scrape latest articles from all or specific websites"""
        results = {
            'total_websites': 0,
            'successful_websites': 0,

            'articles_found': 0,
            'articles_saved': 0
        }
